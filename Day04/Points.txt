abstract method
---------------
abstract method is declared in abstract class that has no implementation, it acts as a blue print that sub classes must implement. This ensures consistency across subclasses while allowing each to provide its own specific behaviour

from abc import ABC, abstractmethod

class Animal(ABC):
  
   def make_sound(self):
      pass

class Dog(Animal):

   def make_sound(self):
      print("bow bow.....")

   def command(self):
      print("sit....")

bruno = Dog()
bruno.command()
